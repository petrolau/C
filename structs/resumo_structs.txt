#include <stdio.h> //printf and scanf
//registro = pacote de variaveis de diferentes tipos.
//campo do registro = cada variavel.
struct dma {
    int dia;
    int mes;
    int ano;
};
struct dma x; // registo x do tipo dma.
struct dma y; // registo y do tipo dma.

//se referindo a um campo de registo(variavel)
x.dia = 31;
x.mes = 12;
x.ano = 2018;

//registro = novo tipo de dados.

typedef struct dma data;
data x,y;
